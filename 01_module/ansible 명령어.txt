vim install.yml

---
- hosts: localhost
  become: yes

  tasks:
    - name: install pip
      yum:
        name:
          - python-pip
          - python3-pip
        state: latest

    - name: install boto
      pip:
        name:
          - boto
          - boto3

  - ap-northeast-2

  instance-state-name: running

#아파치 야믈1 생성
vi httpd.yml


---
- name: Install Apache webserver
  hosts: aws_ec2
  tasks:
    - name: HTTPD is present
      yum:
        name: httpd
        state: present
    - name: Create index.html file
      copy:
        src: /home/ec2-user/index.html
        dest: /var/www/html/index.html
    - name: Start httpd
      service:
        name: httpd
        state: started
        enabled: true

#httpd 야믈 파일 실행
ansible-playbook httpd.yml
